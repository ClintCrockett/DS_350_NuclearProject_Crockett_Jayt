best_brands <- best_brands2 %>%
  pivot_longer(
    cols = starts_with("1/1/"),
    names_to = "year",
    values_to = "amount"
  ) %>%
  mutate(
    year = lubridate::mdy(year) %>% year()
  )

best_brands_19 <- best_brands %>% 
  filter(year == 2019) %>% 
  slice_max(order_by = amount, n = 12) %>% 
  drop_na(amount)
